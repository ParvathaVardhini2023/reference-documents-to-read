
SELECT A.DBNAME,A.TSNAME,A.NAME
         FROM SYSIBM.SYSTABLES A
WHERE A.TSNAME IN
( SELECT B.TSNAME
  FROM SYSIBM.SYSTABLES B
     WHERE B.DBNAME NOT LIKE 'DSNDB0%'
     AND B.TYPE <> 'V'
    GROUP BY TSNAME HAVING COUNT(B.TSNAME) > 1 )
AND A.TSNAME IN
( SELECT D.NAME
  FROM SYSIBM.SYSTABLESPACE D
     WHERE D.SEGSIZE =0
     AND   D.PARTITIONS = 0
 )
AND A.TYPE <> 'V'
--AND A.DBID <> 0
--AND A.OBID <> 0
AND A.DBNAME NOT LIKE 'DSNDB0%'
ORDER BY A.TSNAME
FOR FETCH ONLY WITH UR;

SELECT *
   FROM SYSIBM.SYSTABLESPACE D
      WHERE D.SEGSIZE =0
      AND   D.PARTITIONS = 0
      AND DBNAME NOT LIKE 'DSNDB0%'
FOR FETCH ONLY WITH UR;

 SELECT B.TSNAME
   FROM SYSIBM.SYSTABLES B
      WHERE B.DBNAME NOT LIKE 'DSNDB0%'
      AND B.TYPE <> 'V'
 AND B.TSNAME IN
 ( SELECT D.NAME
   FROM SYSIBM.SYSTABLESPACE D
      WHERE D.SEGSIZE =0
      AND   D.PARTITIONS = 0
  )
     GROUP BY TSNAME HAVING COUNT(B.TSNAME) > 1
 FOR FETCH ONLY WITH UR;
